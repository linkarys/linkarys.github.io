javascript patterns

老生常谈, 做下总结。似乎在ES6之前, js并不算一个完备的面向对象的语言, 本文总结的可能有效期也比较有限, 但是核心思想应该是不变的。当然, 我们并不是为了模式而模式, 要知道很多模式是基于静态语言提出的, 在js中并一定有市场, 也只有在某类模式能够优雅的某种需求时, 才有使用的意义.

单例

使用场景

  • 计数
  • 全局配置
  • 作为模块通信的桥梁

实现

个人觉得以下是个比较好的实现, 可以省略new, 外界没法直接访问和修改instance.

var Singleton = (function() {
  var instance;

  return function single(name) {
    if (instance) return instance;

    // 处理不用new关键字, 直接调用的情况
    if (!(this instanceof single)) {
      return new single(arguments);
    }

    this.name = name;

    instance = this;
  }
}());

工厂

使用场景

在静态语言中用的比较多, 工厂模式有以下目标:

  • 创建相似对象时执行重复操作
  • 编译时不知道具体类型的情况下， 为工厂客户提供一种创建对象的接口(动态语言没有这个烦恼)

实现

var MobileFactory = (function() {
  var constr,
      F = new Function();

  // 使用闭包避免每次调用都要创建代理函数
  return function factory(sys) {
    constr = factory[sys];

    // 如果构造函数不存在, 抛出异常
    if (typeof factory[sys] !== 'function') {
      throw {
        name: 'Error',
        message: 'factory ' + sys + ' is not a valid construtor function'
      }
    }

    // 子类需要继承父类的方法, 这里通过constr.prototype.info判断是否已经继承过,
    // 使用代理F来提高执行效率
    if (typeof constr.prototype.info === 'undefined') {
      F.prototype = MobileFactory.prototype;
      constr.prototype = new F();
      constr.prototype.constructor = constr;
      constr.uber = factory;
    }

    return new constr();
  }
}());

MobileFactory.prototype.info = function() {
  console.log('system info: ' + this.company);
};

// 定义子类构造方法
MobileFactory.ios = function() {
  this.company = 'apple';
};

console.log(MobileFactory.ios.prototype);


var ios = MobileFactory('ios');
ios.info();                                 // system info: apple
console.log(ios instanceof MobileFactory);  // true

var android = MobileFactory('android');     // Error: factory android is not a valid construtor function

有意思的是js内部其实有个“自然工厂”, 那就是Object, 来看下面的例子:

var
    obj = Object(),
    num = Object(1),
    str = Object('str'),
    bln = Object(true);

obj.construtor === Object;                  // true
num.construtor === Number;                  // true
str.construtor === String;                  // true
bln.construtor === Boolean;                 // true

迭代器

在迭代模式中,通常有一个包含某种数据集合的对象。该数据可能存储在一个复杂数据结构内部,而提供一种简单的方法能够访问数据结构中的每一个元素。对象的消费者并不需要知道如何组织数据,所有需要做的就是取出单个数据进行工作。

看了以上的说明, 其实这种模式在js中并没有什么市场。尽管如此, 还是来看一下代码吧:

var agg = (function() {
  var index = 0,
      data = [1,2,3,4,5],
      length = data.length;

  return {
    next: function() {
      if (this.hasNext()) {
        return data[index++];
      }
      return null;
    },
    hasNext: function() {
      return index < length;
    },
    current: function() {
      return data[index];
    },
    rewind: function() {
      index = 0;
    }
  }
}());

装饰者模式

目的：运行时动态添加附加功能到对象中

使用场景

个人觉得在作组件化的时候会比较有用, 因为有些组件间的差异很小, 这个时候你不会希望再建一个类似的组件, 而通过传入大量的配置参数和解决其实对使用者和开发者来说都不算友好, 同时需求的增加, 后期的维护和扩展都将是一个问题

实现

好吧, 下边的代码是我的理解, 不明白这么简单的例子《JavascScript模式》一书, 需要用列表或者继承来实现

var Sale = function(price) {
  this.price = price > 0 || 0;
}

// 增加联邦税
Sale.prototype.fedtax = function() {
  this.price += 20;
  return this;
};

// 增加省级税
Sale.prototype.quebec = function() {
  this.price += 10;
  return this;
};

// 如果是英雄的话, 减10　
Sale.prototype.hero = function() {
  this.price -= 10;
  return this;
};

// 测试
var sale = new Sale();
sale.fedtax().quebec().hero();

console.log(sale);  // { price: 20 }
