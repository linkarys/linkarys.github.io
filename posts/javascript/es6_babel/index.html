Something about Babel & ES6

May 23, 2015 • linkarys

  • javascript
  • es6

Babel可以把ES6代码转换成主流浏览器可以执行的ES5代码(当然转换结果并不完美, 支持也不完整). 好奇babel的转换过程, 所以对某些实现做下简单分析

class

平地一声雷, js终于是告别了没有class的历史！虽可喜, 然而现实依然骨感, 对于es6+, 尚且没有什么浏览器完整支持(截至2015.05.23), 为了提前体验到es6的新特性, 或者说让我们写出的代码更具前瞻性, 我们可能用babel将es6代码转换成es5.

Babel用es5来模拟es6提出的class, 用到了几个关键函数，我们依次来看一下

_createClass

_createClass接受一个构造函数和一个包含key,value的json对象数组, 这些key, value通过Object.definedProperty方法将其添加到构造函数的prototype属性中， 以实现每个构造函数的实例继承这些方法。

var _createClass = (function () {
  function defineProperties(target, props) {
    // 既然都用了Object.defineProperty, 为什么此处不用更简洁的forEach替代?
    for (var i = 0; i < props.length; i++) {
      var descriptor = props[i];

      // 指定属性是否出现在for in操作中(默认为false)
      descriptor.enumerable = descriptor.enumerable || false;

      // 指定该属性能否被删除， 以后是否能改变其descriptor(默认为false)
      descriptor.configurable = true;

      // writable表示定义的属性可写
      if (‘value’ in descriptor) descriptor.writable = true;
      Object.defineProperty(target, descriptor.key, descriptor);
    }
  }
  return function (Constructor, protoProps, staticProps) {
    // 每二个参数数组中定义的key, descriptor会添加到构造函数的prototype中
    if (protoProps) defineProperties(Constructor.prototype, protoProps);

    // 第三个参数数组中定义的key, descriptor会添加到构造函数的自身属性上
    if (staticProps) defineProperties(Constructor, staticProps);
    return Constructor;
  };
})();

_inherits

关于继承, 之前的一篇文章javascript inherit做了较详细的描述

function _inherits(subClass, superClass) {
  if (typeof superClass !== ‘function’ && superClass !== null) {
    throw new TypeError(‘Super expression must either be null or a function, not ‘ + typeof superClass);
  }
  subClass.prototype = Object.create(superClass && superClass.prototype, {
    constructor: {
      value: subClass,
      enumerable: false,
      writable: true,
      configurable: true
    }
  });

// 保留对超类的引用, 另外，如果没有这一步，在现代浏览器中
// Object.getPrototypeOf(subClass)的值将为Function.prototype而不是superClass
  if (superClass) subClass.__proto__ = superClass;
  console.log(Object.getPrototypeOf(subClass));
}

_classCallCheck

比较简单, 应该是防止调用构造函数时通过call, apply, bind之类的方式绑定了不合理的this

function _classCallCheck(instance, Constructor) {
  if (!(instance instanceof Constructor)) {
    throw new TypeError(‘Cannot call a class as a function’);
  }
}

get

在原型链中查找对应方法: 通过Object.getPrototypeof遍历原型链, 通过Object.getOwnPropertyDescriptor获取对应属性值——未找到则继续沿着原型链的方向查找, 这边还考虑到了定义了getter的情况

    var desc = Object.getOwnPropertyDescriptor(object, property);

    if (desc === undefined) {
      var parent = Object.getPrototypeOf(object);
      if (parent === null) {
        return undefined;
      } else {
        _x = parent;
        _x2 = property;
        _x3 = receiver;
        _again = true;

        // 如果有多层循环, 默认continue跳转到所在级别的循环体中,
        // 可以通过continue + label指定跳转的选定的循环体
        continue _function;

    } else if (‘value’ in desc) {
      return desc.value;

function

ES6对于函数的改进主要包括三点

  1. 可以指定默认值了(默认值要位于参数末尾)
  2. rest参数， 用于获取函数的多余参数(将其保存在数组里, 免去之前arguments.slice的操作, 也需要位于参数末尾)
  3. 扩展运算符(spread), 好比rest参数的逆运算 (JavaScript的函数只能返回一个值，如果需要返回多个值，只能返回数组或对象。扩展运算符提供了解决这个问题的一种变通方法)

关于这三点改进, 我们直接看一下前后对比

转换前的代码:

var foo = (a, …b) => {
  console.log(b);
  console.log(…b);
}

var bar = (a, b=3) => a + b;

foo(1, 2, 3, 4, 5);

转换后的代码:

var foo = function foo(a) {
  for (var _len = arguments.length, b = Array(_len > 1 ? _len - 1 : 0), _key = 1; _key < _len; _key++) {
    b[_key - 1] = arguments[_key];
  }

  console.log(b);
  console.log.apply(console, b);
};

var bar = function bar(a) {
  var b = arguments[1] === undefined ? 3 : arguments[1];
  return a + b;
};

foo(1, 2, 3, 4, 5);
// [ 2, 3, 4, 5 ]
// 2 3 4 5
